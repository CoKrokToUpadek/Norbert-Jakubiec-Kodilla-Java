SELECT IL.NAME, I.SUMMARY,I.DESCRIPTION
FROM ISSUES I
JOIN  ISSUESLISTS IL ON IL.ID =I.ISSUESLIST_ID

SELECT I.SUMMARY,I.DESCRIPTION, I.USER_ID_ASSIGNEDTO,U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.USER_ID_ASSIGNEDTO) AS TASKCOUNT
FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY TASKCOUNT
